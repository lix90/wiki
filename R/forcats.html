forcats

forcats 一个用于操作分类变量／因子的R包。

install.packages("forcats")
devtools::install_github("tidyverse/forcats")

forcats::fct_recode() 对指定的因子的水平重新编码。

. %>%
  mutate(
    fct = fct_recode(fct, "level_label2" = "level1", "level_label2" = "level2")
  )
